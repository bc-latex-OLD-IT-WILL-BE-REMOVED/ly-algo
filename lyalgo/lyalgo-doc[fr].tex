\begin{algo}
	\caption{Le crible d'Eratosthène}

    \Data{la liste $L$ des naturels compris de $2$ à $N$.}
    \Result{la liste $P$ des nombres premiers compris entre $2$ et $N$.}
    \BlankLine
    \Actions{
    \For{\range{i}{2}{N}}{
        \If{$i$ n'est pas barré dans la liste $L$}{
            $X \store 4$ soit $4 \putin X$  
            \\
            $x \eqtest 3$ ou $x \eqtest* 3$ 
            \\
            Barrer dans la liste $L$ les mutltiples $ki$ où $k \geq 2$. \Comment*{TTTT}
        }
    }
    \Return{la liste $P$ des nombres non barrés.}
    \\
    \Comment{TTTT}
    }
\end{algo}
